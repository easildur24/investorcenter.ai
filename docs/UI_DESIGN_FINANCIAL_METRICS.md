# UI Design: Financial Metrics Expansion

**Document Version:** 1.0
**Created:** January 26, 2026

---

## Executive Summary

This document outlines the UI design to integrate ~55 new financial metrics into Investor Center while maintaining consistency with existing patterns. The design follows progressive disclosure principles to avoid overwhelming users.

---

## Design Principles

1. **Progressive Disclosure** - Show essential metrics upfront, reveal details on demand
2. **Consistent Patterns** - Reuse existing card, table, and list components
3. **Mobile First** - 2 columns on mobile, 4 on desktop
4. **Contextual Interpretation** - Add badges/colors to help users understand values
5. **Scannable** - Group related metrics, use visual hierarchy

### Reusable MetricCard Component

```tsx
interface MetricCardProps {
  label: string;
  value: string | number;
  suffix?: string;           // 'x', '%', 'days'
  interpretation?: {
    label: string;           // 'Excellent', 'Good', 'Poor'
    color: 'green' | 'yellow' | 'red' | 'neutral';
  };
  comparison?: {
    label: string;           // 'vs Industry', 'vs 5Y Avg'
    value: string;           // '+12%', '-5%'
    direction: 'up' | 'down' | 'neutral';
  };
  description?: string;
  dataSource?: 'fmp' | 'database';
  isAdmin?: boolean;
}

function MetricCard({
  label,
  value,
  suffix,
  interpretation,
  comparison,
  description,
  dataSource,
  isAdmin
}: MetricCardProps) {
  return (
    <div className="bg-ic-bg-secondary rounded-lg p-4">
      <div className="flex items-center justify-between mb-2">
        <span className="text-sm text-ic-text-muted">{label}</span>
        {interpretation && (
          <span className={cn(
            'text-xs px-2 py-0.5 rounded-full',
            interpretationColors[interpretation.color]
          )}>
            {interpretation.label}
          </span>
        )}
      </div>

      <div className="flex items-baseline gap-1">
        <span className="text-2xl font-semibold text-ic-text-primary">
          {formatValue(value)}
        </span>
        {suffix && (
          <span className="text-sm text-ic-text-muted">{suffix}</span>
        )}
        {isAdmin && dataSource && (
          <SourceBadge source={dataSource} />
        )}
      </div>

      {comparison && (
        <div className={cn(
          'text-xs mt-1',
          comparison.direction === 'up' ? 'text-ic-positive' :
          comparison.direction === 'down' ? 'text-ic-negative' :
          'text-ic-text-muted'
        )}>
          {comparison.value} {comparison.label}
        </div>
      )}

      {description && (
        <div className="text-xs text-ic-text-muted mt-1">
          {description}
        </div>
      )}
    </div>
  );
}
```

## Interpretation Logic

### Quality Score Interpretations

```typescript
// Altman Z-Score
function getZScoreInterpretation(score: number): Interpretation {
  if (score > 2.99) return { label: 'Safe Zone', color: 'green', description: 'Low bankruptcy risk' };
  if (score > 1.81) return { label: 'Grey Zone', color: 'yellow', description: 'Moderate risk' };
  return { label: 'Distress Zone', color: 'red', description: 'High bankruptcy risk' };
}

// Piotroski F-Score
function getFScoreInterpretation(score: number): Interpretation {
  if (score >= 8) return { label: 'Strong', color: 'green', description: 'Excellent financial health' };
  if (score >= 5) return { label: 'Average', color: 'yellow', description: 'Moderate financial health' };
  return { label: 'Weak', color: 'red', description: 'Poor financial health' };
}

// PEG Ratio
function getPEGInterpretation(peg: number): Interpretation {
  if (peg < 1) return { label: 'Undervalued', color: 'green', description: 'Growing faster than P/E suggests' };
  if (peg <= 1.5) return { label: 'Fair Value', color: 'neutral', description: 'Reasonably valued' };
  if (peg <= 2) return { label: 'Slightly High', color: 'yellow', description: 'May be overvalued' };
  return { label: 'Overvalued', color: 'red', description: 'Price exceeds growth rate' };
}

// Dividend Payout
function getPayoutInterpretation(ratio: number): Interpretation {
  if (ratio < 30) return { label: 'Very Safe', color: 'green', description: 'Room for dividend growth' };
  if (ratio < 50) return { label: 'Safe', color: 'green', description: 'Sustainable payout' };
  if (ratio < 75) return { label: 'Moderate', color: 'yellow', description: 'Limited room for growth' };
  return { label: 'At Risk', color: 'red', description: 'May not be sustainable' };
}
```
